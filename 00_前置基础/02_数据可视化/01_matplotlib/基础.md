# Matplotlib 绘图

## 基本图表

```python
import matplotlib.pyplot as plt
import numpy as np

# 设置中文字体
plt.rcParams['font.sans-serif'] = ['SimHei', 'Microsoft YaHei']
plt.rcParams['axes.unicode_minus'] = False

# 折线图
x = [1, 2, 3, 4, 5]
y = [2, 4, 6, 8, 10]
plt.plot(x, y)
plt.title("折线图")
plt.xlabel("X轴")
plt.ylabel("Y轴")
plt.show()
```

## 常用图表

```python
# 散点图
plt.scatter(x, y, c='red', s=100, alpha=0.5)

# 柱状图
categories = ["A", "B", "C", "D"]
values = [10, 20, 15, 25]
plt.bar(categories, values, color=['red', 'blue', 'green', 'orange'])

# 饼图
sizes = [15, 30, 45, 10]
labels = ["苹果", "香蕉", "橙子", "葡萄"]
plt.pie(sizes, labels=labels, autopct='%1.1f%%')

# 直方图
data = np.random.randn(1000)
plt.hist(data, bins=30, alpha=0.7)
```

## 子图与布局

```python
fig, axes = plt.subplots(2, 2, figsize=(12, 8))

axes[0, 0].plot(x, y)
axes[0, 1].scatter(x, y)
axes[1, 0].bar(categories, values)
axes[1, 1].pie(sizes, labels=labels)

plt.tight_layout()
plt.show()
```

## 样式设置

```python
plt.figure(figsize=(10, 6))           # 画布大小
plt.grid(True, linestyle='--')        # 网格线
plt.legend(['数据1', '数据2'])         # 图例
plt.xlim(0, 10)                        # X轴范围
plt.ylim(0, 15)                        # Y轴范围
```
